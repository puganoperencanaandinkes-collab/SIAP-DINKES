// ===== USER INFO & ROLE =====
function loadUserInfo() {
  google.script.run
    .withSuccessHandler(function (user) {
      if (!user || !user.isLoggedIn) return;
      currentUser = user;

      document.getElementById('userName').textContent  = user.name;
      document.getElementById('userEmail').textContent = user.email;
      document.getElementById('userAvatar').textContent = user.name.charAt(0).toUpperCase();

      // Badge role di sidebar
      var badge = document.getElementById('userRoleBadge');
      if (user.isAdmin) {
        badge.textContent = '👑 Admin';
        badge.className   = 'role-badge role-admin';
      } else {
        badge.textContent = '👁️ Viewer';
        badge.className   = 'role-badge role-viewer';
      }

      // Badge role di topbar
      var topBadge = document.getElementById('topbarRoleBadge');
      if (user.isAdmin) {
        topBadge.textContent = '👑 Admin';
        topBadge.className   = 'topbar-role role-badge role-admin';
      } else {
        topBadge.textContent = '👁️ Viewer';
        topBadge.className   = 'topbar-role role-badge role-viewer';
      }

      // Tampilkan elemen khusus admin
      applyRoleUI(user.isAdmin);

      // Load dashboard setelah user info siap
      loadDashboard();
    })
    .getCurrentUserWithRole();
}

function applyRoleUI(isAdmin) {
  // Tampilkan menu & elemen admin-only
  document.querySelectorAll('.admin-only').forEach(function (el) {
    el.style.display = isAdmin ? '' : 'none';
  });
}

// ===== NAVIGASI =====
function showPage(page) {
  document.querySelectorAll('.page').forEach(function (p) { p.classList.remove('active'); });
  document.querySelectorAll('.nav-item').forEach(function (n) { n.classList.remove('active'); });

  var el = document.getElementById('page-' + page);
  if (el) el.classList.add('active');

  // Highlight nav item
  var navMap = { dashboard: 0, tambah: 1, daftar: 2, cari: 3, rekap: 4, log: 5 };
  var navItems = document.querySelectorAll('.nav-item');

  // Kalau admin, indeks berbeda karena menu tambah visible
  if (currentUser && currentUser.isAdmin) {
    var adminNavMap = { dashboard: 0, tambah: 1, daftar: 2, cari: 3, rekap: 5, log: 6, kelola: 8 };
    if (adminNavMap[page] !== undefined && navItems[adminNavMap[page]]) {
      navItems[adminNavMap[page]].classList.add('active');
    }
  } else {
    var viewerNavMap = { dashboard: 0, daftar: 1, cari: 2, rekap: 4, log: 5 };
    if (viewerNavMap[page] !== undefined && navItems[viewerNavMap[page]]) {
      navItems[viewerNavMap[page]].classList.add('active');
    }
  }

  var titles = {
    dashboard: 'Dashboard', tambah: 'Tambah Arsip',
    daftar: 'Daftar Arsip', cari: 'Cari Arsip',
    rekap: 'Rekap Tahunan', log: 'Log Aktivitas', kelola: 'Kelola User'
  };
  document.getElementById('topbarTitle').textContent = titles[page] || 'Dashboard';
  currentPage = page;

  if (page === 'daftar') loadDaftarArsip();
  if (page === 'rekap')  loadRekap();
  if (page === 'log')    loadLog();
  if (page === 'kelola') loadDaftarUser();

  closeSidebar();
}

function refreshPage() {
  if (currentPage === 'dashboard') loadDashboard();
  else showPage(currentPage);
}

// ===== DASHBOARD =====
function loadDashboard() {
  google.script.run
    .withSuccessHandler(function (result) {
      if (!result.success) { showToast('Gagal memuat dashboard', 'error'); return; }
      renderStats(result.stats);
      renderChartJenis(result.perJenis);
      renderChartBulanan(result.perBulan);
      renderTerbaru(result.terbaru);
      document.getElementById('tahunChart').textContent = new Date().getFullYear();
      if (result.user) {
        document.getElementById('dashboardGreet').textContent =
          'Selamat datang, ' + result.user.name + ' — ' +
          (result.user.isAdmin ? 'Anda login sebagai Admin' : 'Anda login sebagai Viewer (read-only)');
      }
    })
    .withFailureHandler(function (e) { showToast('Error: ' + e, 'error'); })
    .getStatistikDashboard();
}

function renderChartJenis(perJenis) {
  var colors = ['#4f8ef7','#7c4dff','#22c55e','#f59e0b','#ef4444','#06b6d4','#ec4899','#84cc16'];
  var entries = Object.entries(perJenis);
  var total   = entries.reduce(function (s, e) { return s + e[1]; }, 0);

  if (!entries.length) {
    document.getElementById('chartJenis').innerHTML =
      '<div class="empty-state"><i class="fas fa-chart-pie"></i><p>Belum ada data</p></div>';
    return;
  }

  entries.sort(function (a, b) { return b[1] - a[1]; });

  // Donut SVG
  var cx = 70, cy = 70, r = 55, circumference = 2 * Math.PI * r;
  var svg = '<svg viewBox="0 0 140 140" style="width:130px;height:130px;display:block;margin:0 auto">' +
    '<circle cx="' + cx + '" cy="' + cy + '" r="' + r + '" fill="none" stroke="var(--surface2)" stroke-width="20"/>';
  var offset = 0;
  entries.forEach(function (e, i) {
    var frac = total ? e[1] / total : 0;
    var dash = frac * circumference;
    svg += '<circle cx="' + cx + '" cy="' + cy + '" r="' + r + '" fill="none" ' +
      'stroke="' + colors[i % colors.length] + '" stroke-width="20" ' +
      'stroke-dasharray="' + dash + ' ' + (circumference - dash) + '" ' +
      'stroke-dashoffset="' + (-offset * circumference) + '" ' +
      'transform="rotate(-90 ' + cx + ' ' + cy + ')"/>';
    offset += frac;
  });
  svg += '<text x="' + cx + '" y="' + (cy - 4) + '" text-anchor="middle" fill="var(--text)" ' +
    'font-family="JetBrains Mono" font-size="17" font-weight="bold">' + total + '</text>' +
    '<text x="' + cx + '" y="' + (cy + 13) + '" text-anchor="middle" fill="var(--muted)" font-size="9">TOTAL</text>' +
    '</svg>';

  var legend = '<div class="pie-legend">';
  entries.forEach(function (e, i) {
    var pct = total ? Math.round(e[1] / total * 100) : 0;
    legend += '<div class="pie-item">' +
      '<div class="pie-dot" style="background:' + colors[i % colors.length] + '"></div>' +
      '<div class="pie-name">' + e[0] + '</div>' +
      '<div class="pie-pct">' + pct + '%</div>' +
      '<div class="pie-val">' + e[1] + '</div>' +
    '</div>';
  });
  legend += '</div>';

  document.getElementById('chartJenis').innerHTML = svg + legend;
}

function renderChartBulanan(perBulan) {
  var maxVal = Math.max.apply(null, perBulan.map(function (b) { return b.jumlah; })) || 1;
  var html = '<div class="bar-chart">';
  perBulan.forEach(function (b) {
    var h = Math.max((b.jumlah / maxVal) * 170, b.jumlah > 0 ? 6 : 0);
    html += '<div class="bar-item">' +
      '<div class="bar-fill" style="height:' + h + 'px" data-val="' + b.jumlah + '"></div>' +
      '<div class="bar-label">' + b.bulan + '</div>' +
    '</div>';
  });
  html += '</div>';
  document.getElementById('chartBulanan').innerHTML = html;
}

// ===== TAMBAH ARSIP =====
var selectedFile = null;

function setupFileUpload() {
  var zone  = document.getElementById('uploadZone');
  var input = document.getElementById('fileInput');
  if (!zone || !input) return;

  zone.addEventListener('dragover', function (e) { e.preventDefault(); zone.style.borderColor = 'var(--accent)'; });
  zone.addEventListener('dragleave', function () { zone.style.borderColor = ''; });
  zone.addEventListener('drop', function (e) {
    e.preventDefault(); zone.style.borderColor = '';
    if (e.dataTransfer.files[0]) handleFile(e.dataTransfer.files[0]);
  });
  input.addEventListener('change', function () { if (input.files[0]) handleFile(input.files[0]); });
}

function handleFile(file) {
  if (file.size > 10 * 1024 * 1024) { showToast('Ukuran file melebihi 10MB!', 'error'); return; }
  selectedFile = file;
  var icons = { pdf:'📄', doc:'📝', docx:'📝', xls:'📊', xlsx:'📊', jpg:'🖼️', jpeg:'🖼️', png:'🖼️' };
  var ext = file.name.split('.').pop().toLowerCase();
  var preview = document.getElementById('filePreview');
  preview.classList.remove('hidden');
  preview.innerHTML =
    '<span class="file-icon">' + (icons[ext] || '📎') + '</span>' +
    '<div class="file-info"><div class="file-name">' + esc(file.name) + '</div>' +
    '<div class="file-size">' + fmtSize(file.size) + '</div></div>' +
    '<button class="file-remove" onclick="removeFile()"><i class="fas fa-times"></i></button>';
  document.getElementById('uploadZone').style.display = 'none';
}

function removeFile() {
  selectedFile = null;
  document.getElementById('filePreview').classList.add('hidden');
  document.getElementById('uploadZone').style.display = '';
  document.getElementById('fileInput').value = '';
}

function fmtSize(b) {
  if (b < 1024) return b + ' B';
  if (b < 1048576) return (b / 1024).toFixed(1) + ' KB';
  return (b / 1048576).toFixed(1) + ' MB';
}

function submitArsip() {
  var data = {
    nomorArsip:   gv('nomorArsip'),
    judulArsip:   gv('judulArsip'),
    jenisArsip:   gv('jenisArsip'),
    kategori:     gv('kategori'),
    tanggalArsip: gv('tanggalArsip'),
    pengirim:     gv('pengirim'),
    tujuan:       gv('tujuan'),
    perihal:      gv('perihal'),
    keterangan:   gv('keterangan')
  };

  if (!data.judulArsip || !data.jenisArsip || !data.kategori || !data.tanggalArsip) {
    showToast('Lengkapi field yang wajib diisi (*)', 'error'); return;
  }

  var btn = document.getElementById('btnSubmit');
  btn.disabled = true;
  btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Menyimpan...';

  function send(fileData) {
    google.script.run
      .withSuccessHandler(function (res) {
        btn.disabled = false;
        btn.innerHTML = '<i class="fas fa-save"></i> Simpan Arsip';
        if (res.success) {
          showToast('✅ ' + res.message + ' (ID: ' + res.id + ')', 'success');
          resetForm(); loadDashboard();
        } else {
          showToast('❌ ' + res.message, 'error');
        }
      })
      .withFailureHandler(function (e) {
        btn.disabled = false;
        btn.innerHTML = '<i class="fas fa-save"></i> Simpan Arsip';
        showToast('❌ ' + e, 'error');
      })
      .simpanArsip(data, fileData);
  }

  if (selectedFile) {
    var reader = new FileReader();
    reader.onload = function (e) {
      send({ name: selectedFile.name, content: e.target.result.split(',')[1], mimeType: selectedFile.type });
    };
    reader.readAsDataURL(selectedFile);
  } else {
    send(null);
  }
}

function resetForm() {
  ['nomorArsip','judulArsip','jenisArsip','kategori','tanggalArsip',
   'pengirim','tujuan','perihal','keterangan'].forEach(function (id) {
    var el = document.getElementById(id);
    if (el) el.value = '';
  });
  removeFile();
}

function loadDaftarArsip() {
  document.getElementById('bodyArsip').innerHTML =
    '<tr><td colspan="9"><div class="loading-pulse" style="height:36px;border-radius:6px;margin:6px 0"></div></td></tr>';

  google.script.run
    .withSuccessHandler(function (result) {
      if (!result.success) { showToast('Gagal memuat data', 'error'); return; }
      allArsipData = result.data;
      renderTabelArsip(allArsipData);
    })
    .withFailureHandler(function (e) { showToast('Error: ' + e, 'error'); })
    .getSemuaArsip();
}

function filterDaftar() {
  var kw     = gv('filterKeyword').toLowerCase();
  var jenis  = gv('filterJenis');
  var kat    = gv('filterKategori');
  var status = gv('filterStatus');

  var data = allArsipData.filter(function (r) {
    return (!kw || matchKw(r, kw)) &&
           (!jenis  || r.jenisArsip === jenis) &&
           (!kat    || r.kategori   === kat) &&
           (!status || r.status     === status);
  });
  renderTabelArsip(data);
}

function matchKw(r, kw) {
  return (r.id          && r.id.toLowerCase().includes(kw)) ||
         (r.nomorArsip  && r.nomorArsip.toLowerCase().includes(kw)) ||
         (r.judulArsip  && r.judulArsip.toLowerCase().includes(kw)) ||
         (r.perihal     && r.perihal.toLowerCase().includes(kw)) ||
         (r.pengirim    && r.pengirim.toLowerCase().includes(kw));
}

function renderTabelArsip(data) {
  document.getElementById('tableInfo').textContent =
    'Menampilkan ' + data.length + ' arsip dari ' + allArsipData.length + ' total';

  if (!data.length) {
    document.getElementById('bodyArsip').innerHTML =
      '<tr><td colspan="9" class="text-center" style="padding:40px;color:var(--muted)">Tidak ada data ditemukan</td></tr>';
    return;
  }

  var isAdmin = currentUser && currentUser.isAdmin;

  document.getElementById('bodyArsip').innerHTML = data.map(function (r) {
    var aksiHapus = isAdmin
      ? '<button class="tbl-btn danger" onclick="hapusArsipConfirm(\'' + r.id + '\',\'' + esc(r.judulArsip) + '\')" title="Hapus"><i class="fas fa-trash"></i></button>'
      : '';
    return '<tr>' +
      '<td><span style="font-family:\'JetBrains Mono\';font-size:11px;color:var(--accent)">' + r.id + '</span></td>' +
      '<td style="font-size:12px">' + (r.nomorArsip || '—') + '</td>' +
      '<td><strong style="font-size:13px">' + esc(r.judulArsip) + '</strong>' +
        (r.perihal ? '<div style="font-size:11px;color:var(--muted)">' + esc(r.perihal) + '</div>' : '') +
      '</td>' +
      '<td><span class="badge ' + getBadge(r.jenisArsip) + '">' + r.jenisArsip + '</span></td>' +
      '<td style="font-size:12px;color:var(--muted)">' + r.kategori + '</td>' +
      '<td style="font-size:12px;white-space:nowrap">' + r.tanggalArsip + '</td>' +
      '<td><span class="badge badge-' + r.status.toLowerCase() + '">' + r.status + '</span></td>' +
      '<td>' + (r.fileUrl
        ? '<a href="' + r.fileUrl + '" target="_blank" class="tbl-btn" title="Buka File"><i class="fas fa-eye"></i></a>'
        : '<span style="color:var(--muted);font-size:12px">—</span>') + '</td>' +
      '<td>' +

        aksiHapus +
      '</td>' +
    '</tr>';
  }).join('');
}

// ===== REKAP TAHUNAN =====
function initTahunRekap() {
  var sel = document.getElementById('tahunRekap');
  if (!sel) return;
  var now = new Date().getFullYear();
  for (var y = now; y >= now - 9; y--) {
    sel.innerHTML += '<option value="' + y + '">' + y + '</option>';
  }
}

function loadRekap() {
  var tahun = gv('tahunRekap') || String(new Date().getFullYear());
  document.getElementById('rekapContent').innerHTML =
    '<div class="loading-pulse" style="height:280px;border-radius:12px;margin-top:12px"></div>';

  google.script.run
    .withSuccessHandler(function (result) {
      if (!result.success) return;
      renderRekap(result.data.filter(function (r) { return r.tanggalArsip.includes(tahun); }), tahun);
    })
    .getSemuaArsip();
}

function renderRekap(data, tahun) {
  var perBulan = {};
  for (var m = 1; m <= 12; m++) perBulan[m] = 0;
  data.forEach(function (r) {
    var parts = r.tanggalArsip.split('/');
    if (parts.length >= 2) perBulan[parseInt(parts[1])] = (perBulan[parseInt(parts[1])] || 0) + 1;
  });

  var perJenis = {}, perKategori = {};
  data.forEach(function (r) {
    perJenis[r.jenisArsip]   = (perJenis[r.jenisArsip]   || 0) + 1;
    perKategori[r.kategori]  = (perKategori[r.kategori]  || 0) + 1;
  });

  var bulanNames = ['Januari','Februari','Maret','April','Mei','Juni',
                    'Juli','Agustus','September','Oktober','November','Desember'];
  var colors = ['#4f8ef7','#7c4dff','#22c55e','#f59e0b','#ef4444','#06b6d4','#ec4899'];

  var html =
    '<div class="rekap-grid">' +
      '<div class="rekap-card"><div class="rekap-num" style="color:var(--accent)">' + data.length + '</div>' +
        '<div class="rekap-label">Total Arsip ' + tahun + '</div></div>' +
      '<div class="rekap-card"><div class="rekap-num" style="color:var(--success)">' +
        data.filter(function(r){return r.status==='Aktif';}).length +
        '</div><div class="rekap-label">Arsip Aktif</div></div>' +
      '<div class="rekap-card"><div class="rekap-num" style="color:var(--warning)">' +
        Object.keys(perJenis).length +
        '</div><div class="rekap-label">Jenis Arsip</div></div>' +
    '</div>' +
    '<div class="card"><div class="card-header"><h3><i class="fas fa-table"></i> Detail Per Bulan — ' + tahun + '</h3></div>' +
    '<div class="table-responsive"><table class="data-table"><thead>' +
    '<tr><th>Bulan</th><th>Jumlah</th><th>%</th><th>Visualisasi</th></tr></thead><tbody>';

  bulanNames.forEach(function (bln, i) {
    var jml = perBulan[i + 1] || 0;
    var pct = data.length ? Math.round(jml / data.length * 100) : 0;
    html += '<tr><td><strong>' + bln + '</strong></td>' +
      '<td><span style="font-family:\'JetBrains Mono\';font-size:15px;font-weight:700">' + jml + '</span></td>' +
      '<td><span style="color:var(--muted)">' + pct + '%</span></td>' +
      '<td><div style="background:var(--surface2);border-radius:4px;height:8px;min-width:120px">' +
        '<div style="background:linear-gradient(90deg,var(--accent),var(--accent2));height:100%;border-radius:4px;width:' + pct + '%"></div>' +
      '</div></td></tr>';
  });

  html += '</tbody></table></div></div>';

  // Per Jenis & Kategori
  html += '<div class="charts-grid">';

  var renderPie = function (obj, title) {
    var entries = Object.entries(obj).sort(function(a,b){return b[1]-a[1];});
    var tot = entries.reduce(function(s,e){return s+e[1];},0);
    var out = '<div class="chart-card"><div class="chart-header"><h3>' + title + '</h3></div><div class="pie-legend">';
    entries.forEach(function(e,i){
      out += '<div class="pie-item">' +
        '<div class="pie-dot" style="background:' + colors[i % colors.length] + '"></div>' +
        '<div class="pie-name">' + e[0] + '</div>' +
        '<div class="pie-pct">' + (tot ? Math.round(e[1]/tot*100) : 0) + '%</div>' +
        '<div class="pie-val">' + e[1] + '</div>' +
      '</div>';
    });
    return out + '</div></div>';
  };

  html += renderPie(perJenis, 'Per Jenis Arsip');
  html += renderPie(perKategori, 'Per Kategori');
  html += '</div>';

  document.getElementById('rekapContent').innerHTML = html;
}

function exportRekap() {
  var tahun = gv('tahunRekap') || String(new Date().getFullYear());
  google.script.run
    .withSuccessHandler(function (result) {
      if (!result.success) return;
      var filtered = result.data.filter(function (r) { return r.tanggalArsip.includes(tahun); });
      var headers  = ['ID Arsip','Nomor','Judul','Jenis','Kategori','Tanggal','Pengirim','Perihal','Status','File'];
      var csv = headers.join(',') + '\n';
      filtered.forEach(function (r) {
        csv += [r.id, r.nomorArsip,
          '"' + (r.judulArsip||'').replace(/"/g,'""') + '"',
          r.jenisArsip, r.kategori, r.tanggalArsip,
          '"' + (r.pengirim||'') + '"', '"' + (r.perihal||'') + '"',
          r.status, r.namaFile || ''].join(',') + '\n';
      });
      var blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
      var a = document.createElement('a');
      a.href = URL.createObjectURL(blob);
      a.download = 'Rekap_Arsip_' + tahun + '.csv';
      a.click();
      showToast('Export CSV berhasil!', 'success');
    })
    .getSemuaArsip();
}

function loadDaftarUser() {
  document.getElementById('tabelUser').innerHTML =
    '<div class="loading-pulse" style="height:80px;border-radius:8px"></div>';

  google.script.run
    .withSuccessHandler(function (result) {
      if (!result.success) { showToast('❌ ' + result.message, 'error'); return; }
      if (!result.data.length) {
        document.getElementById('tabelUser').innerHTML =
          '<p style="text-align:center;color:var(--muted);padding:32px">Belum ada user terdaftar</p>';
        return;
      }
      var html = '<div class="table-responsive"><table class="data-table"><thead>' +
        '<tr><th>Email</th><th>Nama</th><th>Role</th><th>Aksi</th></tr></thead><tbody>';
      result.data.forEach(function (u) {
        var isAdm = u.role === 'admin';
        html += '<tr>' +
          '<td>' + esc(u.email) + '</td>' +
          '<td>' + esc(u.nama) + '</td>' +
          '<td><span class="role-badge ' + (isAdm ? 'role-admin' : 'role-viewer') + '">' +
            (isAdm ? '👑 Admin' : '👁️ Viewer') + '</span></td>' +
          '<td><button class="tbl-btn danger" onclick="hapusUserConfirm(\'' + esc(u.email) + '\')">' +
            '<i class="fas fa-trash"></i></button></td>' +
        '</tr>';
      });
      html += '</tbody></table></div>';
      document.getElementById('tabelUser').innerHTML = html;
    })
    .withFailureHandler(function (e) { showToast('❌ ' + e, 'error'); })
    .getDaftarUser();
}

function submitTambahUser() {
  var email = gv('newUserEmail').trim();
  var nama  = gv('newUserNama').trim();
  var role  = gv('newUserRole');
  if (!email) { showToast('Email wajib diisi!', 'error'); return; }

  google.script.run
    .withSuccessHandler(function (result) {
      showToast(result.success ? '✅ ' + result.message : '❌ ' + result.message,
                result.success ? 'success' : 'error');
      if (result.success) {
        document.getElementById('newUserEmail').value = '';
        document.getElementById('newUserNama').value  = '';
        loadDaftarUser();
      }
    })
    .withFailureHandler(function (e) { showToast('❌ ' + e, 'error'); })
    .tambahUser(email, role, nama);
}

// ===== UTILS =====
function getBadge(jenis) {
  if (!jenis) return 'badge-sm-other';
  if (jenis.includes('Masuk')) return 'badge-sm-masuk';
  if (jenis.includes('Keluar')) return 'badge-sm-keluar';
  return 'badge-sm-other';
}

function gv(id) { var el = document.getElementById(id); return el ? el.value : ''; }

function esc(s) {
  return (s || '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function highlight(text, kw) {
  if (!kw || !text) return esc(text || '');
  var re = new RegExp('(' + kw.replace(/[.*+?^${}()|[\]\\]/g,'\\$&') + ')', 'gi');
  return esc(text).replace(re,
    '<mark style="background:rgba(79,142,247,.3);color:var(--text);border-radius:2px;padding:0 2px">$1</mark>');
}

var _toastTimer;
function showToast(msg, type) {
  type = type || 'info';
  var t = document.getElementById('toast');
  t.textContent = msg;
  t.className = 'toast show ' + type;
  clearTimeout(_toastTimer);
  _toastTimer = setTimeout(function () { t.classList.remove('show'); }, 4000);
}
