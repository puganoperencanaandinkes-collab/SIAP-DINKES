<script>
function doLogin() {
  var btn = document.getElementById('btnLogin');
  btn.disabled = true;
  btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Memverifikasi akun...';
  hideError();

  // doLogin() di server membuat token sesi dan mengembalikan URL redirect
  // Token disimpan di PropertiesService sehingga doGet bisa baca role tanpa Session
  google.script.run
    .withSuccessHandler(function (res) {
      if (res && res.success) {
        document.getElementById('loginForm').style.display = 'none';
        var ls = document.getElementById('loadingState');
        ls.style.display = 'block';
        document.getElementById('loadingText').textContent =
          'Halo ' + res.name + '! Membuka halaman ' +
          (res.role === 'admin' ? 'Admin' : 'Viewer') + '...';

        // Redirect ke URL dengan token — doGet akan baca token & tampilkan halaman
        setTimeout(function () {
          window.top.location.href = res.url;
        }, 700);
      } else {
        var msg = (res && res.message)
          ? res.message
          : 'Akun Google tidak terdeteksi. Pastikan sudah login Google di browser ini.';
        showError(msg);
        btn.disabled = false;
        btn.innerHTML = '<i class="fab fa-google"></i> Lanjutkan dengan Google';
      }
    })
    .withFailureHandler(function (e) {
      showError('Terjadi kesalahan: ' + (e.message || e));
      btn.disabled = false;
      btn.innerHTML = '<i class="fab fa-google"></i> Lanjutkan dengan Google';
    })
    .doLogin();
}

function showError(msg) {
  var box = document.getElementById('errorBox');
  document.getElementById('errorMsg').textContent = msg;
  box.style.display = 'flex';
}
function hideError() {
  document.getElementById('errorBox').style.display = 'none';
}
