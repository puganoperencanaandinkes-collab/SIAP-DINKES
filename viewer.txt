window.onload = function () {
  google.script.run
    .withSuccessHandler(function (u) {
      if (!u || !u.isLoggedIn) return;
      document.getElementById('userAv').textContent = u.name.charAt(0).toUpperCase();
      document.getElementById('userNm').textContent = u.name;
    })
    .getCurrentUser();
};

// ===== SEARCH =====
function doSearch() {
  var kw     = document.getElementById('keyword').value.trim();
  var filter = {
    jenisArsip: document.getElementById('fJenis').value,
    kategori:   document.getElementById('fKategori').value,
    tahun:      document.getElementById('fTahun').value
  };

  document.getElementById('results').innerHTML =
    '<div class="loading-pulse" style="height:80px;margin-bottom:10px"></div>' +
    '<div class="loading-pulse" style="height:80px;margin-bottom:10px"></div>' +
    '<div class="loading-pulse" style="height:80px"></div>';

  google.script.run
    .withSuccessHandler(function (res) {
      if (!res.success) { showToast('Gagal mencari: ' + res.message, 'error'); return; }
      renderResults(res.data, kw, res.total);
    })
    .withFailureHandler(function (e) { showToast('Error: ' + e, 'error'); })
    .cariArsip(kw, filter);
}

function renderResults(data, kw, total) {
  if (!data || !data.length) {
    document.getElementById('results').innerHTML =
      '<div class="empty-state"><i class="fas fa-folder-open"></i>' +
      '<p>Tidak ada arsip yang cocok dengan <strong>"' + esc(kw) + '"</strong></p></div>';
    return;
  }

  var html = '<p class="result-meta">Ditemukan <strong>' + total +
    '</strong> arsip' + (kw ? ' untuk kata kunci <strong>"' + esc(kw) + '"</strong>' : '') + '</p>';

  data.forEach(function (r) {
    var hasFile = r.fileUrl && r.fileUrl !== '';
    html +=
      '<div class="result-card">' +
        '<div class="result-body">' +
          '<div class="result-top">' +
            '<div class="result-info">' +
              '<div class="result-badges">' +
                '<span class="b-id">' + r.id + '</span>' +
                '<span class="badge ' + badgeCls(r.jenisArsip) + '">' + r.jenisArsip + '</span>' +
                '<span class="badge ' + (r.status==='Aktif'?'b-aktif':'b-arsip') + '">' + r.status + '</span>' +
              '</div>' +
              '<div class="result-title">' + hl(r.judulArsip, kw) + '</div>' +
              (r.perihal ? '<div class="result-perihal">' + hl(r.perihal, kw) + '</div>' : '') +
              '<div class="result-meta-row">' +
                '<span><i class="fas fa-calendar-alt"></i>' + r.tanggalArsip + '</span>' +
                (r.pengirim ? '<span><i class="fas fa-user"></i>' + esc(r.pengirim) + '</span>' : '') +
                (r.nomorArsip ? '<span><i class="fas fa-hashtag"></i>' + esc(r.nomorArsip) + '</span>' : '') +
                '<span><i class="fas fa-tag"></i>' + r.kategori + '</span>' +
              '</div>' +
            '</div>' +
            '<div class="result-actions">' +
              '<button class="btn-detail" onclick="lihatDetail(\'' + r.id + '\')">' +
                '<i class="fas fa-info-circle"></i> Detail' +
              '</button>' +
              (hasFile
                ? '<a class="btn-download" href="' + r.fileUrl + '" target="_blank">' +
                    '<i class="fas fa-download"></i> Unduh' +
                  '</a>'
                : '<span class="btn-download disabled"><i class="fas fa-ban"></i> No File</span>'
              ) +
            '</div>' +
          '</div>' +
        '</div>' +
      '</div>';
  });

  document.getElementById('results').innerHTML = html;
}

function closeModal() {

// ===== UTILS =====
function badgeCls(j) {
  if (!j) return 'b-other';
  if (j.includes('Masuk'))  return 'b-masuk';
  if (j.includes('Keluar')) return 'b-keluar';
  return 'b-other';
}
function esc(s) {
  return (s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
function hl(text, kw) {
  if (!kw || !text) return esc(text||'');
  var re = new RegExp('(' + kw.replace(/[.*+?^${}()|[\]\\]/g,'\\$&') + ')', 'gi');
  return esc(text).replace(re, '<mark>$1</mark>');
}

var _tt;
function showToast(msg, type) {
  var t = document.getElementById('toast');
  t.textContent = msg; t.className = 'toast show ' + (type||'info');
  clearTimeout(_tt); _tt = setTimeout(function(){ t.classList.remove('show'); }, 4000);
}
